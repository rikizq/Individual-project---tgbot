def search_music(query, limit=SEARCH_LIMIT):
    url = "https://itunes.apple.com/search"
    params = {
        "term": query,
        "entity": "song",
        "limit": limit,
        "country": "RU",
        "lang": "ru_ru"
    }

        tracks = []

                "trackName": e["im:name"]["label"],
                "artistName": e["im:artist"]["label"],
